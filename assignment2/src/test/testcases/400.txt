                    Function: fact
                        Parameter: n
                        Body:
                            If n == 0 Then
                                Return 1;
                            Else
                                Return n * fact (n - 1);
                            EndIf.
                        EndBody.

                    Function: sum
                        Parameter: n
                        Body:
                            If n == 0 Then
                                Return 0;
                            Else
                                Return n + sum (n - 1);
                            EndIf.
                        EndBody.

                        Body:
                            x = 10;
                            fact (x);
                            sum (x);
                        EndBody.
